    <select id="type">
      <option value="Book">Book</option>

    </select>

    <label for="subCategory">Sub-category:</label>
    <input type="text" id="subCategory">

    <label>
      <input type="checkbox" id="isCompleted"> Marked as Read/Watched
    </label>

    <button onclick="addMedia()">Add to List</button>
  </div>

  <div class="media-list">
    <h2>Tracked Media</h2>
    <table>
      <thead>
        <tr>
          <th>Name</th>
          <th>Category</th>
          <th>Sub-category</th>
          <th>Status</th>
        </tr>
      </thead>
      <tbody id="mediaTableBody"></tbody>
    </table>
  </div>

  <script>
    const mediaList = [];

    function addMedia() {
      const name = document.getElementById("mediaName").value.trim();
      const type = document.getElementById("type").value;
      const sub = document.getElementById("subCategory").value.trim();
      const done = document.getElementById("isCompleted").checked;

      if (!name) return alert("Please enter a name.");

      mediaList.push({ name, type, sub, done });
      renderTable();

      document.getElementById("mediaName").value = "";
      document.getElementById("subCategory").value = "";
      document.getElementById("isCompleted").checked = false;
    }

    function renderTable() {
      const body = document.getElementById("mediaTableBody");
      body.innerHTML = "";
      mediaList.forEach(media => {
        const row = `<tr>
          <td>${media.name}</td>
          <td>${media.type}</td>
          <td>${media.sub}</td>
          <td>${media.done ? "✅ Done" : "❌ Not Yet"}</td>
        </tr>`;
        body.innerHTML += row;
      });
    }
  </script>
